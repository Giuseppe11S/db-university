1. Selezionare tutti gli studenti iscritti al Corso di Laurea in Economia

SELECT students.*
FROM students
JOIN degrees ON students.degree_id = degrees.id
WHERE degrees.name = 'Corso di Laurea in Economia';

2. Selezionare tutti i Corsi di Laurea Magistrale del Dipartimento di
Neuroscienze
SELECT dpt.name, drg.name, drg.level
FROM degrees AS drg
JOIN departments AS dpt ON dpt.id = drg.department_id
WHERE dpt.name = 'Dipartimento di Neuroscienze'
  AND drg.level = 'magistrale';

3. Selezionare tutti i corsi in cui insegna Fulvio Amato (id=44)
SELECT courses.*
FROM courses
JOIN course_teacher ON courses.id = course_teacher.course_id
JOIN teachers ON teachers.id = course_teacher.teacher_id
WHERE teachers.id = 44;

4. Selezionare tutti gli studenti con i dati relativi al corso di laurea a cui
sono iscritti e il relativo dipartimento, in ordine alfabetico per cognome e nome

5. Selezionare tutti i corsi di laurea con i relativi corsi e insegnanti
SELECT students.id, students.name AS nome_stud,
students.surname AS cognome_stud,
degrees.name AS corso_laurea,
departments.name AS dpt
FROM students
JOIN degrees ON students.degree_id = degrees.id
JOIN departments ON degrees.department_id = departments.id
ORDER BY cognome_stud, nome_stud;

6. Selezionare tutti i docenti che insegnano nel Dipartimento di
Matematica (54)
SELECT degrees.name AS corso_laurea,
courses.name AS crs,
teachers.name AS docente_nome,
teachers.surname AS docente_cognome
FROM degrees
JOIN courses ON degrees.id = courses.degree_id
JOIN course_teacher ON courses.id = course_teacher.course_id
JOIN teachers ON teachers.id = course_teacher.teacher_id
ORDER BY corso_laurea, crs;


SELECT DISTINCT t.surname AS cognome_teacher, t.name AS nome_teacher , dpt.name AS NOME_DIPARTIMENTO
FROM departments AS dpt
JOIN degrees AS drg ON drg.department_id = dpt.id
JOIN courses AS crs ON crs.degree_id = drg.id
JOIN course_teacher AS ct ON ct.course_id = crs.id
JOIN teachers AS t ON t.id = ct.teacher_id
WHERE dpt.name = 'Dipartimento di Matematica'
ORDER BY t.surname, t.name;
