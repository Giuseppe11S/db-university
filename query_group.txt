
SELECT YEAR(`enrolment_date`) AS anno_iscrizione,
COUNT(*) AS totale_studenti
FROM `students`
GROUP BY `anno_iscrizione`;

SELECT `office_address` AS sede_ufficio,
COUNT(*) AS totale_docenti
FROM `teachers`
GROUP BY `office_address`;


SELECT `exam_id`,
AVG(`vote`) AS `media_voti`
FROM `exam_student`
GROUP BY `exam_id`;


SELECT `department_id`,
COUNT(*) AS `totale_corsi_laurea`
FROM `degrees`
GROUP BY `department_id`;
